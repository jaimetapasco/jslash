1 (function() {
  2 	
  3 /** jslash shapes module
  4  * @namespace
  5  */
  6 /* TODO: possible refactorization with an abstract Object Shape */
  7 /* TODO: add shadow effects */
  8 	
  9 jslash.shapes = {};
 10 
 11 /** Drawable Rectangle
 12  *  @this {jslash.shapes.Rectangle}
 13  *  @constructor
 14  *  @param {number} x 
 15  *  @param {number} y
 16  *  @param {number} width 
 17  *  @param {number} height
 18  *  @param {jslash.Color} color
 19  *  @property {number} x 
 20  *  @property {number} y
 21  *  @property {number} width 
 22  *  @property {number} height
 23  *  @property {jslash.Color} color
 24  */
 25 
 26 jslash.shapes.Rectangle = function(x,y,width,height,color) {
 27   this.x = x;
 28   this.y = y;
 29   this.width = width;
 30   this.height = height;
 31   this.color = jslash.isDefined(color) ? color : new jslash.Color(0,0,0);
 32   this._filled = true;
 33 };
 34 
 35 /** Object will draw only the shape border with a borderSize.
 36  * @this {jslash.shapes.Rectangle}
 37  * @param {number} strokeSize
 38  */
 39 
 40 jslash.shapes.Rectangle.prototype.stroke = function(strokeSize) {
 41   this._strokeSize = strokeSize;
 42   this._filled = false;
 43 };
 44 
 45 /** Object will fill the shape with the defined color
 46  * @this {jslash.shapes.Rectangle}
 47  */
 48 jslash.shapes.Rectangle.prototype.fill = function() {
 49   this._filled = true;
 50 };
 51 
 52 /** Draw the rectangle on the canvas context.
 53  * @this {jslash.shapes.Rectangle}
 54  * @param {Context2D} ctx
 55  */
 56 
 57 jslash.shapes.Rectangle.prototype.draw = function(ctx) {
 58   if (this._filled) {
 59     ctx.fillStyle = this.color.toString();
 60     ctx.fillRect(this.x,this.y,this.width, this.height);
 61   }
 62   else {
 63     ctx.strokeStyle = this.color.toString();
 64     ctx.lineWidth = this._strokeSize;
 65     ctx.strokeRect(this.x, this.y, this.width, this.height);
 66   }
 67 };
 68 
 69 /** Returns an object with a Sizeable interface
 70  * @this {jslash.shapes.Rectangle}
 71  * @return {jslash.Sizeable}
 72  */
 73 
 74 jslash.shapes.Rectangle.prototype.toSizeable = function() {
 75 	var that = this;
 76 	return { width: function() { return that.width; },
 77 		     height: function() { return that.height; }};
 78 };
 79 
 80 /** Drawable Circle
 81  *  @this {jslash.shapes.Circle}
 82  *  @constructor
 83  *  @param {number} cx The center x coordinate 
 84  *  @param {number} cy The center y coordinate
 85  *  @param {number} radius 
 86  *  @param {jslash.Color} color
 87  *  @property {number} center_x The center x coordinate
 88  *  @property {number} center_y The center y coordinate
 89  *  @property {number} radius 
 90  *  @property {jslash.Color} color
 91  */
 92 
 93 jslash.shapes.Circle = function(cx,cy,radius,color) {
 94   this.center_x = cx;
 95   this.center_y = cy;
 96   this.radius = radius;
 97   this.color = jslash.isDefined(color) ? color : new jslash.Color(0,0,0);
 98   this._filled = true;
 99 };
100 
101 /** Object will draw only the shape border with a borderSize.
102  * @this {jslash.shapes.Circle}
103  * @param {number} strokeSize
104  */
105 
106 jslash.shapes.Circle.prototype.stroke = function(strokeSize) {
107   this._strokeSize = strokeSize;
108   this._filled = false;
109 };
110 
111 /** Object will fill the shape with the defined color
112  * @this {jslash.shapes.Circle}
113  */
114 
115 jslash.shapes.Circle.prototype.fill = function() {
116   this._filled = true;
117 };
118 
119 /** Draws the circle on a canvas
120  * @this {jslash.shapes.Circle}
121  * @param {Context2D} ctx
122  */
123 
124 jslash.shapes.Circle.prototype.draw = function(ctx) {
125   ctx.beginPath();
126   ctx.arc(this.center_x,this.center_y,this.radius,0,2*Math.PI,0);
127   if (this._filled) {
128     ctx.fillStyle = this.color.toString();
129     ctx.fill();
130   }
131   else {
132     ctx.strokeStyle = this.color.toString();
133     ctx.lineWidth = this._strokeSize;
134     ctx.stroke();
135   }
136   ctx.closePath();
137 };

139 /** Drawable Line
140  *  @this {jslash.shapes.Line}
141  *  @constructor
142  *  @param {jslash.Point} from The starting point.
143  *  @param {jslash.Point} to The finishing point.
144  *  @param {jslash.Color} [color] The line color.
145  *  @param {number} [width] The line width.
146  *  @property {jslash.Point} from The starting point.
147  *  @property {jslash.Point} to The finishing point.
148  *  @property {jslash.Color} color The line color.
149  *  @property {number} width The line width.
150  */
151 
152 jslash.shapes.Line = function(from,to,color,width) {
153 	this.from = from;
154 	this.to = to;
155 	this.color = jslash.isDefined(color) ? color : new jslash.Color(0,0,0);
156 	this.width = jslash.isDefined(width) ? width : 1;
157 }
158 
159 /** Draws the line on a canvas
160  * @this {jslash.shapes.Line}
161  * @param {Context2D} ctx
162  */
163 
164 jslash.shapes.Line.prototype.draw = function(ctx) {
165 	ctx.save();
166 	ctx.fillStyle = this.color.toString();
167 	ctx.lineWidth = this.width;
168 	ctx.beginPath();
169 	ctx.moveTo(this.from.x,this.from.y);
170 	ctx.lineTo(this.to.x, this.to.y);
171 	ctx.closePath();
172 	ctx.stroke();
173 	ctx.restore();
174 };
